Get a list of banners.

This endpoint lets you get a list of banners

Example request:
curl --request GET \
    --get "http://localhost:8000/api/v1/banner?limit=50" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --header "API-KEY: KHGCDevelopment2023"
const url = new URL(
    "http://localhost:8000/api/v1/banner"
);

const params = {
    "limit": "50",
};
Object.keys(params)
    .forEach(key => url.searchParams.append(key, params[key]));

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
    "API-KEY": "KHGCDevelopment2023",
};

fetch(url, {
    method: "GET",
    headers,
}).then(response => response.json());
